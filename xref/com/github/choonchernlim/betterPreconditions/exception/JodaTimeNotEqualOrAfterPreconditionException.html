6   /**
7    * This exception is thrown if the first value is not equal to or after second value.
8    */
9   public class JodaTimeNotEqualOrAfterPreconditionException extends PreconditionException {
10  
11      /**
12       * Constructs a new exception with an error message.
13       *
14       * @param firstBaseLocalValue  First base local value
15       * @param firstBaseLocalLabel  First base local label
16       * @param secondBaseLocalValue Second base local value
17       * @param secondBaseLocalLabel Second base local label
18       */
19      public JodaTimeNotEqualOrAfterPreconditionException(final BaseLocal firstBaseLocalValue,
20                                                          final String firstBaseLocalLabel,
21                                                          final BaseLocal secondBaseLocalValue,
22                                                          final String secondBaseLocalLabel) {
23          super(String.format("%s [ %s ] must be equal to or after %s [ %s ]",
24                              firstBaseLocalLabel,
25                              firstBaseLocalValue,
26                              secondBaseLocalLabel,
27                              secondBaseLocalValue));
28      }
29  }
