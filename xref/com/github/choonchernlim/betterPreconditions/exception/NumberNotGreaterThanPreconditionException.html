5   /**
6    * This exception is thrown if the first value is not greater than second value.
7    */
8   public class NumberNotGreaterThanPreconditionException extends PreconditionException {
9   
10      /**
11       * Constructs a new exception with an error message.
12       *
13       * @param firstValue  First value
14       * @param firstLabel  First label
15       * @param secondValue Second value
16       * @param secondLabel Second label
17       */
18      public NumberNotGreaterThanPreconditionException(final Number firstValue,
19                                                       final String firstLabel,
20                                                       final Number secondValue,
21                                                       final String secondLabel) {
22          super(String.format("%s [ %s ] must be greater than %s [ %s ]",
23                              firstLabel,
24                              firstValue,
25                              secondLabel,
26                              secondValue));
27      }
28  }
