1   package com.github.choonchernlim.betterPreconditions.core;
2   
3   import com.github.choonchernlim.betterPreconditions.exception.ObjectEqualPreconditionException;
4   import com.github.choonchernlim.betterPreconditions.exception.ObjectNotEqualPreconditionException;
5   import com.github.choonchernlim.betterPreconditions.exception.ObjectNotNullPreconditionException;
6   import com.github.choonchernlim.betterPreconditions.exception.ObjectNotSameTypePreconditionException;
7   import com.github.choonchernlim.betterPreconditions.exception.ObjectNullPreconditionException;
8   import com.github.choonchernlim.betterPreconditions.exception.ObjectSameTypePreconditionException;
9   import com.github.choonchernlim.betterPreconditions.exception.StringBlankPreconditionException;
10  import static com.github.choonchernlim.betterPreconditions.preconditions.PreconditionFactory.expect;
11  import static com.google.common.base.Strings.nullToEmpty;
12  import com.google.common.collect.Lists;
13  
14  import java.util.List;
15  
16  /**
17   * Abstract class for all preconditions.
18   */
19  public abstract class Preconditions<C, V> {
20  
21      /**
22       * Value to be checked.
23       */
24      protected final V value;
25  
26      /**
27       * Label for the value.
28       */
29      protected final String label;
30  
31      /**
32       * All assertions to be executed.
33       */
34      private final List<Assertion> assertions;
35  
36      /**
37       * Mutable field to indicate whether the upcoming assertion is negated.
38       */
39      private boolean isNegated;
40  
41      /**
42       * Creates a new instance.
43       *
44       * @param value Value
45       * @param label Label
46       */
47      protected Preconditions(final V value, final String label) {
48          // label is not empty... cannot use `expect(label).not().toBeBlank().check()` to prevent
49          // infinite recursion calls.
50          if (nullToEmpty(label).trim().isEmpty()) {
51              throw new StringBlankPreconditionException(label, "Label");
52          }
53  
54          this.value = value;
55          this.label = label;
56          this.assertions = Lists.newArrayList();
57          this.isNegated = false;
58      }
59  
60      /**
61       * Enables negation.
62       *
63       * @return Current instance
64       */
65      @SuppressWarnings("unchecked")
66      public final C not() {
67          this.isNegated = true;
68          return (C) this;
69      }
70  
71      /**
72       * Ensures the object is null.
73       *
74       * @return Current instance
75       */
76      @SuppressWarnings("unchecked")
77      public final C toBeNull() {
78          customMatcher(new Matcher<V>() {
79              @Override
80              public boolean match(final V value, final String label) {
81                  return value == null;
82              }
83  
84              @Override
85              public PreconditionException getException(final V value, final String label) {
86                  return new ObjectNotNullPreconditionException(value, label);
87              }
88  
89              @Override
90              public PreconditionException getNegatedException(final V value, final String label) {
91                  return new ObjectNullPreconditionException(value, label);
92              }
93          });
94  
95          return (C) this;
96      }
97  
98      /**
99       * Ensures given base local is equal to expected value.
100      *
101      * @see Preconditions#toBeEqual(Object, String)
102      */
103     public C toBeEqual(final V expectedValue) {
104         return toBeEqual(expectedValue, "Expected Value");
105     }
106 
107     /**
108      * Ensures given value is equal to expected value.
109      *
110      * @param expectedValue Second value
111      * @param expectedLabel Second label
112      * @return Current instance
113      */
114     @SuppressWarnings("unchecked")
115     public C toBeEqual(final V expectedValue, final String expectedLabel) {
116         expectValueLabelToExist(expectedValue, expectedLabel, "Expected Label");
117 
118         customMatcher(new Matcher<V>() {
119             @Override
120             public boolean match(final V givenValue, final String givenLabel) {
121                 return expect(givenValue, givenLabel).not().toBeNull().check().equals(expectedValue);
122             }
123 
124             @Override
125             public PreconditionException getException(final V givenValue, final String givenLabel) {
126                 return new ObjectNotEqualPreconditionException(givenValue, givenLabel, expectedValue, expectedLabel);
127             }
128 
129             @Override
130             public PreconditionException getNegatedException(final V givenValue, final String givenLabel) {
131                 return new ObjectEqualPreconditionException(givenValue, givenLabel, expectedValue, expectedLabel);
132             }
133         });
134 
135         return (C) this;
136     }
137 
138     /**
139      * Ensures given base local is equal to expected value.
140      *
141      * @see Preconditions#toBeSameType(Object, String)
142      */
143     public C toBeSameType(final V expectedValue) {
144         return toBeSameType(expectedValue, "Expected Value");
145     }
146 
147     /**
148      * Ensures given value has same class type as expected value.
149      *
150      * @param expectedValue Second value
151      * @param expectedLabel Second label
152      * @return Current instance
153      */
154     @SuppressWarnings("unchecked")
155     public C toBeSameType(final V expectedValue, final String expectedLabel) {
156         expectValueLabelToExist(expectedValue, expectedLabel, "Expected Label");
157 
158         customMatcher(new Matcher<V>() {
159             @Override
160             public boolean match(final V givenValue, final String givenLabel) {
161                 return expect(givenValue, givenLabel)
162                         .not().toBeNull().check()
163                         .getClass().equals(expectedValue.getClass());
164             }
165 
166             @Override
167             public PreconditionException getException(final V givenValue, final String givenLabel) {
168                 return new ObjectNotSameTypePreconditionException(givenValue, givenLabel, expectedValue, expectedLabel);
169             }
170 
171             @Override
172             public PreconditionException getNegatedException(final V givenValue, final String givenLabel) {
173                 return new ObjectSameTypePreconditionException(givenValue, givenLabel, expectedValue, expectedLabel);
174             }
175         });
176 
177         return (C) this;
178     }
179 
180     /**
181      * Adds new assertion.
182      *
183      * @param matcher Matcher
184      * @return Current instance
185      */
186     @SuppressWarnings("unchecked")
187     public final C customMatcher(final Matcher<V> matcher) {
188         assertions.add(new Assertion<V>(matcher, isNegated, value, label));
189 
190         // after creating a new assertion, reset the negation
191         this.isNegated = false;
192 
193         return (C) this;
194     }
195 
196     /**
197      * Returns value if all assertions pass.
198      */
199     public final V check() {
200         for (Assertion assertion : assertions) {
201             assertion.run();
202         }
203 
204         return value;
205     }
206 
207     /**
208      * Ensures value not be null and label not be blank.
209      *
210      * @param value        Value
211      * @param label        Label
212      * @param defaultLabel Default label
213      */
214     protected final void expectValueLabelToExist(final Object value, final String label, final String defaultLabel) {
215         expect(label, defaultLabel).not().toBeBlank().check();
216         expect(value, label).not().toBeNull().check();
217     }
218 
219     /**
220      * Ensures value not null and same type.
221      *
222      * @param givenValue    Given label
223      * @param givenLabel    Given label
224      * @param expectedValue Expected value
225      * @param expectedLabel Expected label
226      */
227     protected final void expectNotNullAndSameType(final V givenValue,
228                                                   final String givenLabel,
229                                                   final V expectedValue,
230                                                   final String expectedLabel) {
231 
232         expect(expectedValue, expectedLabel)
233                 .not().toBeNull()
234                 .toBeSameType(givenValue, givenLabel)
235                 .check();
236     }
237 }
