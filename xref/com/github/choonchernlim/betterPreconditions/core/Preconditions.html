1   package com.github.choonchernlim.betterPreconditions.core;
2   
3   import com.github.choonchernlim.betterPreconditions.exception.ObjectEqualPreconditionException;
4   import com.github.choonchernlim.betterPreconditions.exception.ObjectNotEqualPreconditionException;
5   import com.github.choonchernlim.betterPreconditions.exception.ObjectNotNullPreconditionException;
6   import com.github.choonchernlim.betterPreconditions.exception.ObjectNullPreconditionException;
7   import com.github.choonchernlim.betterPreconditions.exception.StringBlankPreconditionException;
8   import static com.github.choonchernlim.betterPreconditions.preconditions.PreconditionFactory.expect;
9   import static com.google.common.base.Strings.nullToEmpty;
10  import com.google.common.collect.Lists;
11  
12  import java.util.List;
13  
14  /**
15   * Abstract class for all preconditions.
16   */
17  public abstract class Preconditions<C, V> {
18  
19      /**
20       * Value to be checked.
21       */
22      protected final V value;
23  
24      /**
25       * Label for the value.
26       */
27      protected final String label;
28  
29      /**
30       * All assertions to be executed.
31       */
32      private final List<Assertion> assertions;
33  
34      /**
35       * Mutable field to indicate whether the upcoming assertion is negated.
36       */
37      private boolean isNegated;
38  
39      /**
40       * Creates a new instance.
41       *
42       * @param value Value
43       * @param label Label
44       */
45      protected Preconditions(final V value, final String label) {
46          // label is not empty... cannot use `expect(label).not().toBeBlank().check()` to prevent
47          // infinite recursion calls.
48          if (nullToEmpty(label).trim().isEmpty()) {
49              throw new StringBlankPreconditionException(label, "Label");
50          }
51  
52          this.value = value;
53          this.label = label;
54          this.assertions = Lists.newArrayList();
55          this.isNegated = false;
56      }
57  
58      /**
59       * Enables negation.
60       *
61       * @return Current instance
62       */
63      @SuppressWarnings("unchecked")
64      public final C not() {
65          this.isNegated = true;
66          return (C) this;
67      }
68  
69      /**
70       * Ensures the object is null.
71       *
72       * @return Current instance
73       */
74      @SuppressWarnings("unchecked")
75      public final C toBeNull() {
76          customMatcher(new Matcher<V>() {
77              @Override
78              public boolean match(final V value, final String label) {
79                  return value == null;
80              }
81  
82              @Override
83              public PreconditionException getException(final V value, final String label) {
84                  return new ObjectNotNullPreconditionException(value, label);
85              }
86  
87              @Override
88              public PreconditionException getNegatedException(final V value, final String label) {
89                  return new ObjectNullPreconditionException(value, label);
90              }
91          });
92  
93          return (C) this;
94      }
95  
96      /**
97       * Ensures given base local is equal to expected value.
98       *
99       * @see Preconditions#toBeEqual(Object)
100      */
101     public C toBeEqual(final V expectedValue) {
102         return toBeEqual(expectedValue, "Expected Value");
103     }
104 
105     /**
106      * Ensures given value is equal to expected value.
107      *
108      * @param expectedValue Second value
109      * @param expectedLabel Second label
110      * @return Current instance
111      */
112     @SuppressWarnings("unchecked")
113     public C toBeEqual(final V expectedValue, final String expectedLabel) {
114         expectValueLabelToExist(expectedValue, expectedLabel, "Expected Label");
115 
116         customMatcher(new Matcher<V>() {
117             @Override
118             public boolean match(final V givenValue, final String givenLabel) {
119                 return expect(givenValue, givenLabel).not().toBeNull().check().equals(expectedValue);
120             }
121 
122             @Override
123             public PreconditionException getException(final V givenValue, final String givenLabel) {
124                 return new ObjectNotEqualPreconditionException(givenValue, givenLabel, expectedValue, expectedLabel);
125             }
126 
127             @Override
128             public PreconditionException getNegatedException(final V givenValue, final String givenLabel) {
129                 return new ObjectEqualPreconditionException(givenValue, givenLabel, expectedValue, expectedLabel);
130             }
131         });
132 
133         return (C) this;
134     }
135 
136     /**
137      * Adds new assertion.
138      *
139      * @param matcher Matcher
140      * @return Current instance
141      */
142     @SuppressWarnings("unchecked")
143     public final C customMatcher(final Matcher<V> matcher) {
144         assertions.add(new Assertion<V>(matcher, isNegated, value, label));
145 
146         // after creating a new assertion, reset the negation
147         this.isNegated = false;
148 
149         return (C) this;
150     }
151 
152     /**
153      * Returns value if all assertions pass.
154      */
155     public final V check() {
156         for (Assertion assertion : assertions) {
157             assertion.run();
158         }
159 
160         return value;
161     }
162 
163     /**
164      * Ensures value not be null and label not be blank.
165      *
166      * @param value        Value
167      * @param label        Label
168      * @param defaultLabel Default label
169      */
170     protected final void expectValueLabelToExist(final Object value, final String label, final String defaultLabel) {
171         expect(label, defaultLabel).not().toBeBlank().check();
172         expect(value, label).not().toBeNull().check();
173     }
174 }
