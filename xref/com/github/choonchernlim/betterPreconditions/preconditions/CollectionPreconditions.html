3   import com.github.choonchernlim.betterPreconditions.core.Matcher;
4   import com.github.choonchernlim.betterPreconditions.core.PreconditionException;
5   import com.github.choonchernlim.betterPreconditions.core.Preconditions;
6   import com.github.choonchernlim.betterPreconditions.exception.CollectionEmptyPreconditionException;
7   import com.github.choonchernlim.betterPreconditions.exception.CollectionNotEmptyPreconditionException;
8   import static com.github.choonchernlim.betterPreconditions.preconditions.PreconditionFactory.expect;
9   
10  import java.util.Collection;
11  
12  /**
13   * Collection related preconditions.
14   */
15  public class CollectionPreconditions extends Preconditions<CollectionPreconditions, Collection> {
16      /**
17       * Package constructor.
18       *
19       * @param value Value
20       * @param label Label
21       */
22      CollectionPreconditions(final Collection value, final String label) {
23          super(value, label);
24      }
25  
26      /**
27       * Ensures collection is empty.
28       *
29       * @return Current instance
30       */
31      public CollectionPreconditions toBeEmpty() {
32          return customMatcher(new Matcher<Collection>() {
33              @Override
34              public boolean match(final Collection value, final String label) {
35                  return expect(value, label).not().toBeNull().check().isEmpty();
36              }
37  
38              @Override
39              public PreconditionException getException(final Collection value, final String label) {
40                  return new CollectionNotEmptyPreconditionException(value, label);
41              }
42  
43              @Override
44              public PreconditionException getNegatedException(Collection value, String label) {
45                  return new CollectionEmptyPreconditionException(value, label);
46              }
47          });
48      }
49  }
